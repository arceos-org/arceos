1use alloc::collections::VecDeque;
2
3use crate::event::{Callback, IpiEvent};
4
5/// A queue of IPI events.
6///
7/// It internally uses a `VecDeque` to store the events, make it
8/// possible to pop these events using FIFO order.
9pub struct IpiEventQueue {
10    events: VecDeque<IpiEvent>,
11}
12
13impl IpiEventQueue {
14    /// Create a new empty timer list.
15    pub fn new() -> Self {
16        Self {
17            events: VecDeque::new(),
18        }
19    }
20
21    /// Whether there is no event.
22    #[allow(dead_code)]
23    #[inline]
24    pub fn is_empty(&self) -> bool {
25        self.events.is_empty()
26    }
27
28    /// Push a new event into the queue.
29    pub fn push(&mut self, src_cpu_id: usize, callback: Callback) {
30        self.events.push_back(IpiEvent {
31            src_cpu_id,
32            callback,
33        });
34    }
35
36    /// Try to pop the latest event that exists in the queue.
37    ///
38    /// Return `None` if no event is available.
39    #[must_use]
40    pub fn pop_one(&mut self) -> Option<(usize, Callback)> {
41        if let Some(e) = self.events.pop_front() {
42            Some((e.src_cpu_id, e.callback))
43        } else {
44            None
45        }
46    }
47}
48
49impl Default for IpiEventQueue {
50    fn default() -> Self {
51        Self::new()
52    }
53}
